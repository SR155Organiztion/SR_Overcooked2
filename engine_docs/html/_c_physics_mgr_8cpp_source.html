222bool CPhysicsMgr::Check_AABB_Collision(IPhysics* _pPhys, IPhysics* _pOtherPhys)
223{
224 if (
225 !_pPhys->Get_Opt()->bApplyCollision
226 || !_pOtherPhys->Get_Opt()->bApplyCollision
227 ) return false;
228
229 _vec3* aMin = _pPhys->Get_NextMinBox();
230 _vec3* aMax = _pPhys->Get_NextMaxBox();
231 _vec3* bMin = _pOtherPhys->Get_NextMinBox();
232 _vec3* bMax = _pOtherPhys->Get_NextMaxBox();
233
234 if (IsSameBox(bMin, _pOtherPhys->Get_MinBox()) && IsSameBox(bMax, _pOtherPhys->Get_MaxBox())) {
235 const float fEpsilon = 0.05f;
236 bMin->x -= fEpsilon; bMax->x += fEpsilon;
237 bMin->z -= fEpsilon; bMax->z += fEpsilon;
238 }
239
240 BOOL bCollides =
241 (aMin->x <= bMax->x && aMax->x >= bMin->x) &&
242 (aMin->y <= bMax->y && aMax->y >= bMin->y) &&
243 (aMin->z <= bMax->z && aMax->z >= bMin->z);
244
245 return bCollides;

264void CPhysicsMgr::Resolve_Collision(IPhysics* _pSelf, IPhysics* _pOther, CTransform* _pTransform)
265{
266 CGameObject* pOtherObj = nullptr;
267
268 for (CGameObject* pCandidate : m_physicsList)
269 {
270 if (dynamic_cast<IPhysics*>(pCandidate) == _pOther)
271 {
272 pOtherObj = pCandidate;
273 break;
274 }
275 }
276
277 if (!pOtherObj) return;
278
279 CTransform* pOtherTransform = dynamic_cast<CTransform*>(pOtherObj->Get_Component(ID_DYNAMIC, L"Com_Transform"));
280 if (!pOtherTransform) return;
281
282 _vec3 dir = _pTransform->m_vInfo[INFO_POS] - _pTransform->m_vPrevPos;
283 _vec3* pVel = _pTransform->Get_Velocity();
284
285 if (_pOther->Get_Opt()->bApplyKnockBack)
286 {
287 _vec3 pushDir = dir;
288 D3DXVec3Normalize(&pushDir, &pushDir);
289
290 _vec3 pushForce = pushDir * 300.0f;
291 }
292
293
294 if (abs(dir.y) > abs(dir.x) && abs(dir.y) > abs(dir.z))
295 {
296 _pTransform->m_vInfo[INFO_POS].y = _pTransform->m_vPrevPos.y;
297 pVel->y = 0.f;
298 _pSelf->Set_IsGround(true);
299 _pSelf->Set_GravityElapsed(0.f);
300
301 _pTransform->m_bBlocked[1] = true;
302
303 if (dir.y < 0.f || pOtherTransform->m_vInfo[INFO_POS].y <= 0.01f)
304 {
305 _vec3 stationPos = pOtherTransform->m_vInfo[INFO_POS];
306 _float stationH = pOtherTransform->Get_Scale().y * 0.5f;
307 _float selfH = _pTransform->Get_Scale().y * 0.5f;
308
309 _vec3 snapPos = stationPos;
310 snapPos.y = 0.f + stationH + selfH;
311
312 _pTransform->Set_Pos(snapPos.x, snapPos.y, snapPos.z);
313 *pVel = _vec3(0.f, 0.f, 0.f);
314 }
315 }
316 else
317 {
318 if (abs(dir.x) > abs(dir.z))
319 {
320 _pTransform->m_vInfo[INFO_POS].x = _pTransform->m_vPrevPos.x;
321 pVel->x = 0.f;
322 _pTransform->m_bBlocked[0] = true; // X방향 차단
323 }
324 else
325 {
326 _pTransform->m_vInfo[INFO_POS].z = _pTransform->m_vPrevPos.z;
327 pVel->z = 0.f;
328 _pTransform->m_bBlocked[2] = true; // Z방향 차단
329 }
330 }
331
332 if (_pSelf->Get_Opt()->bApplyBouncing)
333 {
334 pVel->y *= -0.5f;
335 }
336 else if (_pSelf->Get_Opt()->bApplyRolling)
337 {
338 pVel->x *= _pSelf->Get_Opt()->fDeceleration;
339 pVel->z *= _pSelf->Get_Opt()->fDeceleration;
340 }
341}

350_bool CPhysicsMgr::Check_AnyCollision(CTransform* _pTransform, const _vec3& _vTargetPos)
351{
352 if (!_pTransform)
353 return false;
354 CGameObject* pSelfObj = nullptr;
355 for (CGameObject* pObj : m_physicsList)
356 {
357 CTransform* pCandidateTransform = dynamic_cast<CTransform*>(pObj->Get_Component(ID_DYNAMIC, L"Com_Transform"));
358 if (pCandidateTransform == _pTransform)
359 {
360 pSelfObj = pObj;
361 break;
362 }
363 }
364 if (!pSelfObj) return false;
365
366 IPhysics* pSelfPhysics = dynamic_cast<IPhysics*>(pSelfObj);
367 if (!pSelfPhysics) return false;
368
369 CVIBuffer* pSelfBuffer = dynamic_cast<CVIBuffer*>(pSelfObj->Get_Component(ID_STATIC, L"Com_Buffer"));
370 if (!pSelfBuffer) return false;
371
372 _vec3 extent = (pSelfBuffer->Get_MaxBox() - pSelfBuffer->Get_MinBox()) * 0.5f;
373 _vec3 selfMin = _vTargetPos - extent;
374 _vec3 selfMax = _vTargetPos + extent;
375
376 for (CGameObject* pOther : m_physicsList)
377 {
378 if (pOther == pSelfObj) continue;
379
380 IPhysics* pOtherPhysics = dynamic_cast<IPhysics*>(pOther);
381 if (!pOtherPhysics) continue;
382
383 _vec3* otherMin = pOtherPhysics->Get_NextMinBox();
384 _vec3* otherMax = pOtherPhysics->Get_NextMaxBox();
385
386 if (selfMin.x <= otherMax->x && selfMax.x >= otherMin->x &&
387 selfMin.y <= otherMax->y && selfMax.y >= otherMin->y &&
388 selfMin.z <= otherMax->z && selfMax.z >= otherMin->z)
389 {
390 return true;
391 }
392 }
393
394 return false;
395}
