24_float CCalculator::Compute_HeightOnTerrain(const _vec3* pPos, const _vec3* pTerrainVtxPos,
25 const _ulong& dwCntX,
26 const _ulong& dwCntZ,
27 const _ulong& dwVtxItv)
28{
29 _ulong dwIndex = _ulong(pPos->z / dwVtxItv) * dwCntX + _ulong(pPos->x / dwVtxItv);
30
31 _float fWidth = (pPos->x - pTerrainVtxPos[dwIndex + dwCntX].x) / dwVtxItv;
32 _float fHeight = (pTerrainVtxPos[dwIndex + dwCntX].z - pPos->z) / dwVtxItv;
33
34 D3DXPLANE Plane;
35
36 // 오른쪽 위
37 if (fWidth > fHeight)
38 {
39 D3DXPlaneFromPoints(&Plane,
40 &pTerrainVtxPos[dwIndex + dwCntX],
41 &pTerrainVtxPos[dwIndex + dwCntX + 1],
42 &pTerrainVtxPos[dwIndex + 1]);
43 }
44
45 // 왼쪽 아래
46 else
47 {
48 D3DXPlaneFromPoints(&Plane,
49 &pTerrainVtxPos[dwIndex + dwCntX],
50 &pTerrainVtxPos[dwIndex + 1],
51 &pTerrainVtxPos[dwIndex]);
52 }
53
54 // ax + by + cz + d
55 //
56 // -by = ax + cz + d
57 //
58 // y = (-ax - cz - d) / b;
59
60 return (-Plane.a * pPos->x - Plane.c * pPos->z - Plane.d) / Plane.b;
61}

63_vec3 CCalculator::Picking_OnTerrain(HWND hWnd, const CTerrainTex* pTerrainBufferCom, const CTransform* pTerrainTransformCom)
64{
65 POINT ptMouse{};
66
67 GetCursorPos(&ptMouse);
68 ScreenToClient(hWnd, &ptMouse);
69
70 // 뷰포트 -> 투영
71
72 _vec3 vMousePos;
73
74 D3DVIEWPORT9 ViewPort;
75 ZeroMemory(&ViewPort, sizeof(D3DVIEWPORT9));
76
77 m_pGraphicDev->GetViewport(&ViewPort);
78
79 vMousePos.x = ptMouse.x / (ViewPort.Width * 0.5f) - 1.f;
80 vMousePos.y = ptMouse.y / -(ViewPort.Height * 0.5f) + 1.f;
81 vMousePos.z = 0.f;
82
83 // 투영 -> 뷰 스페이스
84 D3DXMATRIX matProj;
85 m_pGraphicDev->GetTransform(D3DTS_PROJECTION, &matProj);
86 D3DXMatrixInverse(&matProj, 0, &matProj);
87 D3DXVec3TransformCoord(&vMousePos, &vMousePos, &matProj);
88
89 // 뷰 스페이스 -> 월드
90
91 D3DXMATRIX matView;
92 m_pGraphicDev->GetTransform(D3DTS_VIEW, &matView);
93 D3DXMatrixInverse(&matView, 0, &matView);
94
95 _vec3 vRayPos{ 0.f, 0.f, 0.f };
96 _vec3 vRayDir = vMousePos - vRayPos;
97 D3DXVec3Normalize(&vRayDir, &vRayDir);
98
99 D3DXVec3TransformCoord(&vRayPos, &vRayPos, &matView);
100 D3DXVec3TransformNormal(&vRayDir, &vRayDir, &matView);
101
102 // 월드 -> 로컬
103 _matrix matWorld;
104 pTerrainTransformCom->Get_World(&matWorld);
105 D3DXMatrixInverse(&matWorld, 0, &matWorld);
106
107 D3DXVec3TransformCoord(&vRayPos, &vRayPos, &matWorld);
108 D3DXVec3TransformNormal(&vRayDir, &vRayDir, &matWorld);
109
110 _ulong dwVtxIdx[3]{};
111 _float fU(0.f), fV(0.f), fDist(0.f);
112
113 const VTXTEX* pTerrainVtx = pTerrainBufferCom->Get_VTXTEX();
114
115 for (_ulong i = 0; i < VTXCNTZ - 1; ++i)
116 {
117 for (_ulong j = 0; j < VTXCNTX - 1; ++j)
118 {
119 _ulong dwIndex = i * VTXCNTX + j;
120
121 // 오른쪽 위
122 dwVtxIdx[0] = dwIndex + VTXCNTX;
123 dwVtxIdx[1] = dwIndex + VTXCNTX + 1;
124 dwVtxIdx[2] = dwIndex + 1;
125
126 // V1 + U(V2 - V1) + V(V3 - V1) // 함수의 매개 변수로 삽입한 순서에 근거한 결과
127
128 if (D3DXIntersectTri(&pTerrainVtx[dwVtxIdx[1]].vPosition,
129 &pTerrainVtx[dwVtxIdx[0]].vPosition,
130 &pTerrainVtx[dwVtxIdx[2]].vPosition,
131 &vRayPos, &vRayDir,
132 &fU, &fV, &fDist))
133 {
134 return _vec3(pTerrainVtx[dwVtxIdx[1]].vPosition.x + (fU * (pTerrainVtx[dwVtxIdx[0]].vPosition.x - pTerrainVtx[dwVtxIdx[1]].vPosition.x)),
135 0.f,
136 pTerrainVtx[dwVtxIdx[1]].vPosition.z + (fV * (pTerrainVtx[dwVtxIdx[2]].vPosition.z - pTerrainVtx[dwVtxIdx[1]].vPosition.z)));
137 }
138
139
140 // 왼쪽 아래
141
142 dwVtxIdx[0] = dwIndex + VTXCNTX;
143 dwVtxIdx[1] = dwIndex + 1;
144 dwVtxIdx[2] = dwIndex;
145
146 // V1 + U(V2 - V1) + V(V3 - V1)
147
148 if (D3DXIntersectTri(&pTerrainVtx[dwVtxIdx[2]].vPosition,
149 &pTerrainVtx[dwVtxIdx[1]].vPosition,
150 &pTerrainVtx[dwVtxIdx[0]].vPosition,
151 &vRayPos, &vRayDir,
152 &fU, &fV, &fDist))
153 {
154 return _vec3(pTerrainVtx[dwVtxIdx[2]].vPosition.x + (fU * (pTerrainVtx[dwVtxIdx[1]].vPosition.x - pTerrainVtx[dwVtxIdx[2]].vPosition.x)),
155 0.f,
156 pTerrainVtx[dwVtxIdx[2]].vPosition.z + (fV * (pTerrainVtx[dwVtxIdx[0]].vPosition.z - pTerrainVtx[dwVtxIdx[2]].vPosition.z)));
157 }
158 }
159
160 }
161
162 return _vec3(0.f, 0.f, 0.f);
163}
