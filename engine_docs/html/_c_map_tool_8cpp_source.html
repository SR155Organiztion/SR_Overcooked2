27HRESULT CMapTool::Load_Json()

29 m_mapJson.clear();

31 std::ifstream file("../Bin/Data/SaveData.json");
32
33 if (!file.is_open()) {
34 MSG_BOX("파일 열기 실패");
35 return E_FAIL;
36 }
37
38 {
39 json j;
40 file >> j;
41
42 // 순회하며 각 요소를 map<string, S_STAGE>로 변환
43 for (auto it = j.begin(); it != j.end(); ++it)
44 {
45 const std::string& stageName = it.key();
46 const json& stageJson = it.value();
47
48 S_STAGE stage = stageJson.get<S_STAGE>();
49 m_mapJson[stageName] = stage;
50 }
51 file.close();
52 }
53
54 return S_OK;
55}
